class Solution {
    public int getSecondLargest(int[] arr) {
        // Code Here
        int largest=arr[0];
        int secondLargest=-1;
        
        
        for(int i:arr){
            if(i>largest){
                secondLargest=largest;
                largest=i;
            }
            else{
                if(i>secondLargest && i!=largest){
                    secondLargest=i;
                }
            }
        }
        return secondLargest;
        
    }
}
